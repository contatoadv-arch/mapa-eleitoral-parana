// ... (mantenha o CSS e HTML anterior, mude apenas a função de processamento) function processarDados(csv, filtroCargo) { const linhas = csv.split('\n'); if (linhas.length < 2) return; // 1. MAPEAMENTO DE COLUNAS (O banco pode estar em qualquer ordem) const cabecalho = linhas[0].split(',').map(h => h.replace(/"/g, '').trim().toUpperCase()); const col = { turno: cabecalho.indexOf('NR_TURNO'), cargo: cabecalho.indexOf('DS_CARGO'), mun: cabecalho.indexOf('NM_MUNICIPIO'), cand: cabecalho.indexOf('NM_VOTAVEL'), part: cabecalho.indexOf('SG_PARTIDO'), votos: cabecalho.indexOf('QT_VOTOS'), sit: cabecalho.indexOf('DS_SIT_TOT_TURNO') // Situação (Eleito, Suplente, etc) }; const novoMapa = {}; // 2. FILTRAGEM RIGOROSA for (let i = 1; i < linhas.length; i++) { const linha = linhas[i].split(',').map(c => c.replace(/"/g, '').trim()); if (linha.length < 5) continue; const turno = linha[col.turno]; const cargoPlanilha = linha[col.cargo]?.toUpperCase(); const municipio = linha[col.mun]?.toUpperCase(); const candidato = linha[col.cand]?.toUpperCase(); const partido = linha[col.part]?.toUpperCase(); const votos = parseInt(linha[col.votos]) || 0; // --- O FILTRO DE ISOLAMENTO --- // Regra A: Apenas Primeiro Turno (Isso evita dobrar votos de quem foi para o 2º) if (turno !== "1") continue; // Regra B: Apenas o cargo que clicamos (Limpa a mistura de Prefeito/Vereador) if (cargoPlanilha !== filtroCargo) continue; // Regra C: Evitar Candidatos Nulos/Brancos no ranking principal (opcional) if (candidato.includes("BRANCO") || candidato.includes("NULO")) continue; // 3. AGRUPAÇÃO if (!novoMapa[municipio]) novoMapa[municipio] = {}; if (!novoMapa[municipio][candidato]) { novoMapa[municipio][candidato] = { nome: candidato, part: partido, votos: 0 }; } novoMapa[municipio][candidato].votos += votos; } // 4. FINALIZAÇÃO DO CACHE ISOLADO dadosAtivos = {}; for (let m in novoMapa) { dadosAtivos[m] = Object.values(novoMapa[m]).sort((a,b) => b.votos - a.votos); } }
